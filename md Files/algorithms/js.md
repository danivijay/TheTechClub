# JS Algorithms


### String

Reverse a string:
```js
function reverse(str) {
  return str.split("").reverse().join("");
}
```

Find longest word:
```js
function findLongestWord(str) {
  var words = str.split(" ");
  str = words.sort(function(a, b) {
    return b.length - a.length;
  })[0];
  return str;
}
```

### Numbers

Factorialize:
```js
function factorialize(num) {
  if(num === 0) return 1;
  return num * factorialize(num-1);
}
```

### Sorting

Bubble Sort:
```js
function bub(arr) {
  for (var j = 0; j < arr.length; j++) {
    for (var i = 0; i < (arr.length - j); i++) {
      if (arr[i] > arr[i + 1]) {
        var a = arr[i]
        var b = arr[i + 1]
        arr[i] = b
        arr[i + 1] = a
      }
    }
  }
  return arr;
}
```

Quicksort:
```js
var quick = function(arr, condition) {
  if(arr.length < 2) {
    return arr;
  };
  var pivot = arr[Math.floor(Math.random()*arr.length)];
  var left = [];
  var equal = [];
  var right = [];
  for(var i = 0; i < arr.length; i++) {
    var item = arr[i];
    var sortable = condition(item, pivot);
    if(item === pivot) {
      equal.push(item)
    } else if(sortable) {
      left.push(item);
    } else {
      right.push(item);
    }
  }
  return quick(left, condition).concat(equal, quick(right, condition));
}
```

### Search

Binary Search
```js
var binarySearch = function(query, arr) {
  var start = 0;
  var end = arr.length - 1;
  while(start <= end) {
    var index = Math.round((start + end) / 2);
    var item = arr[index];
    if(query === item) {
      return index;
    } else if(query > item) {
      start = index + 1;
    } else {
      end = index - 1;
    }
  }
  return null;
}
binarySearch(3, [1, 2, 3, 4, 5]);
// => 2
```
